### Installation
```
go get -u github.com/toel-app/biteship-go
```

### Usage
```go
package main

import (
	"fmt"
	"github.com/toel-app/biteship"
	"log"
)

func main() {
	b := biteship.New(
		biteship.WithSecret("somesecret"),
		biteship.WithUrl("https://api.biteship.com"),
	)

	response, err := b.GetCouriers()
	if err != nil {
		log.Panic(err)
	}

	fmt.Println(response.Couriers)
}
```
